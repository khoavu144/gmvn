## III. DATABASE SCHEMA - ATHLETE PROFILE

### Minimal but Complete

```sql
-- Main athlete profile
CREATE TABLE athlete_profiles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) UNIQUE,
    
    -- BASIC INFO
    athlete_name VARCHAR(255),
    athlete_number INT, -- Jersey number
    profile_image_url VARCHAR(500),
    cover_image_url VARCHAR(500),
    bio_short VARCHAR(500),
    bio_long TEXT, -- Markdown support
    
    -- SPORT DETAILS
    sport_type VARCHAR(100), -- 'boxing', 'mma', 'football', etc
    sport_level VARCHAR(50), -- 'amateur', 'semi-pro', 'professional', 'elite'
    competition_division VARCHAR(255), -- 'Welterweight', 'U21', etc
    
    -- TEAM/LOCATION
    current_team VARCHAR(255),
    current_club VARCHAR(255),
    country VARCHAR(100),
    city VARCHAR(255),
    
    -- PHYSICAL
    height_cm INT,
    weight_kg DECIMAL(5,2),
    
    -- PRIMARY COACH
    primary_coach_id UUID REFERENCES trainers(id),
    
    -- SOCIAL METRICS
    following_count INT DEFAULT 0,
    fans_count INT DEFAULT 0,
    
    -- VISIBILITY
    is_public BOOLEAN DEFAULT true,
    
    -- CUSTOM FIELDS (for admin flexibility)
    custom_fields JSONB, -- [
                         -- {name: "Reach", value: "180cm", icon: "arms"},
                         -- {name: "Stance", value: "Orthodox", icon: ""},
                         -- ]
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_user_id,
    INDEX idx_sport_type
);

-- Achievements
CREATE TABLE athlete_achievements (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    athlete_id UUID NOT NULL REFERENCES athlete_profiles(id),
    
    achievement_title VARCHAR(255), -- "National Championship Gold"
    organization VARCHAR(255), -- "Vietnam Boxing Federation"
    achievement_year INT,
    achievement_date DATE,
    location VARCHAR(255),
    
    medal_type VARCHAR(50), -- 'gold', 'silver', 'bronze', 'winner'
    image_url VARCHAR(500),
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_athlete_id,
    INDEX idx_achievement_year
);

-- Competition matches
CREATE TABLE athlete_competitions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    athlete_id UUID NOT NULL REFERENCES athlete_profiles(id),
    
    match_date DATE,
    opponent_name VARCHAR(255),
    opponent_country VARCHAR(100),
    
    result VARCHAR(50), -- 'win', 'loss', 'draw', 'dq'
    result_detail VARCHAR(255), -- "KO Round 3", "Decision UD"
    
    location VARCHAR(255),
    competition_name VARCHAR(255), -- Tournament name
    
    video_url VARCHAR(500),
    notes TEXT,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_athlete_id,
    INDEX idx_match_date
);

-- Media/Gallery
CREATE TABLE athlete_media (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    athlete_id UUID NOT NULL REFERENCES athlete_profiles(id),
    
    media_url VARCHAR(500),
    media_type VARCHAR(50), -- 'training', 'competition', 'event', 'portrait'
    caption TEXT,
    
    uploaded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    display_order INT,
    
    INDEX idx_athlete_id
);

-- Ranking
CREATE TABLE athlete_rankings (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    athlete_id UUID NOT NULL REFERENCES athlete_profiles(id),
    
    ranking_organization VARCHAR(255), -- "National", "SEA Games", "IBF", etc
    sport_type VARCHAR(100),
    division VARCHAR(100),
    
    rank_position INT,
    points DECIMAL(10,2),
    
    as_of_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_athlete_id,
    INDEX idx_as_of_date
);

-- Sponsorships
CREATE TABLE athlete_sponsorships (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    athlete_id UUID NOT NULL REFERENCES athlete_profiles(id),
    
    sponsor_name VARCHAR(255),
    sponsor_logo_url VARCHAR(500),
    sponsor_type VARCHAR(100), -- 'equipment', 'apparel', 'nutrition'
    
    start_year INT,
    end_year INT,
    is_current BOOLEAN,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_athlete_id
);

-- Media mentions/press
CREATE TABLE athlete_press (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    athlete_id UUID NOT NULL REFERENCES athlete_profiles(id),
    
    article_title VARCHAR(255),
    publication_name VARCHAR(255),
    publication_date DATE,
    article_url VARCHAR(500),
    
    description TEXT,
    image_url VARCHAR(500),
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_athlete_id,
    INDEX idx_publication_date
);
```
